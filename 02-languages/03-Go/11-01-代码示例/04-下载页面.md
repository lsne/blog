import (
	"fmt"
	"io/ioutil"
	"net/http"
	"os"
	"time"
)

func main() {
	themes := []int{70, 134, 233}
	for _, theme := range themes {
		for i := 1; i <= 1000; i++ {
			time.Sleep(2 * time.Second)
			go curlget(theme, i)
		}
	}
}

func curlget(theme, page int) {

	url := fmt.Sprintf("https://www.duoduoli.info/forum-%d-%d.html", theme, page)

	// 创建一个不加载图片的自定义 HTTP 客户端
	client := &http.Client{
		CheckRedirect: func(req *http.Request, via []*http.Request) error {
			return http.ErrUseLastResponse
		},
	}

	// 发送 GET 请求
	resp, err := client.Get(url)
	if err != nil {
		fmt.Println("请求失败:", err)
		return
	}
	defer resp.Body.Close()

	// 读取响应内容
	body, err := ioutil.ReadAll(resp.Body)
	if err != nil {
		fmt.Println("读取响应失败:", err)
		return
	}

	os.WriteFile(fmt.Sprintf("log/%d-%d.log", theme, page), body, 0644)

}
